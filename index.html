// ============================================================ // SCRIPT GOOGLE APPS SCRIPT - Suivi Convoyages K Controles // À coller dans Extensions > Apps Script de votre Google Sheet // ============================================================ const SHEET_NAME = "Convoyages"; const TARIF = 3; // € net par convoyage aller/retour const ADMIN_PIN = "1234"; // ⚠️ Changez ce code PIN ! const DRIVE_FOLDER_NAME = "Photos Convoyages K Controles"; // Dossier Google Drive // Colonnes (15 au total) : // 1:ID | 2:Date | 3:Heure prise | 4:Plaque | 5:Format | 6:Client // 7:Statut prise | 8:Photo prise (lien) | 9:Note prise // 10:Statut clôture | 11:Note clôture | 12:Photo clôture (lien) // 13:Heure clôture | 14:Coût (€) | 15:Convoyeur // // ⚠️ TARIF : seul le convoyeur nommé "PAUL" est rémunéré 3€/convoyage. // Les autres convoyeurs ont un coût de 0€ (salariés). // ============================================================ // INITIALISATION // ============================================================ const SPREADSHEET_ID = "18OvOLnr3fWBnlTc_LV2O45hoSuBY4sqVrwO7u51ric4"; function initSheet() { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); const headers = [ "ID", "Date", "Heure prise en charge", "Plaque", "Format", "Client", "Statut prise", "Photo prise (lien Drive)", "Note prise", "Statut clôture", "Note clôture", "Photo clôture (lien Drive)", "Heure clôture", "Coût (€)", "Convoyeur" ]; sheet.getRange(1, 1, 1, headers.length).setValues([headers]) .setBackground("#3A3A3A").setFontColor("#FFFFFF").setFontWeight("bold"); sheet.setFrozenRows(1); sheet.setColumnWidth(8, 220); sheet.setColumnWidth(12, 220); } return sheet; } // Récupère ou crée le dossier Google Drive pour les photos function getPhotoFolder() { const folders = DriveApp.getFoldersByName(DRIVE_FOLDER_NAME); if (folders.hasNext()) return folders.next(); return DriveApp.createFolder(DRIVE_FOLDER_NAME); } // Upload une photo base64 vers Drive, retourne l'URL publique function uploadPhoto(base64, mime, filename) { const folder = getPhotoFolder(); const blob = Utilities.newBlob( Utilities.base64Decode(base64), mime, filename ); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return file.getUrl(); } // ============================================================ // REQUÊTES GET (rapports admin) // ============================================================ function doGet(e) { const action = e.parameter.action; if (action === "rapport_jour") return jsonResponse(getRapportJour(e.parameter.date)); if (action === "rapport_mois") return jsonResponse(getRapportMois(e.parameter.mois, e.parameter.annee)); return jsonResponse({ error: "Action inconnue" }); } // ============================================================ // REQUÊTES POST (actions convoyeur) // ============================================================ function doPost(e) { const data = JSON.parse(e.postData.contents); const action = data.action; if (action === "start") return jsonResponse(startConvoyage(data)); if (action === "close") return jsonResponse(closeConvoyage(data)); if (action === "verify_pin") return jsonResponse({ ok: data.pin === ADMIN_PIN }); return jsonResponse({ error: "Action inconnue" }); } // ============================================================ // DÉMARRER UN CONVOYAGE // ============================================================ function startConvoyage(data) { const sheet = initSheet(); const now = new Date(); const id = Utilities.getUuid().substring(0, 8).toUpperCase(); const tz = Session.getScriptTimeZone(); const dateStr = Utilities.formatDate(now, tz, "dd/MM/yyyy"); const heureStr = Utilities.formatDate(now, tz, "HH:mm"); const statutPrise = data.statutPrise || "OK"; // "OK" ou "PROBLÈME" const plaque = data.plaque ? data.plaque.toUpperCase() : ""; // Upload photo prise en charge si fournie let photoPriseUrl = ""; if (data.photo && data.photoMime) { try { const filename = `${plaque}_prise_${dateStr.replace(/\//g,'-')}_${heureStr.replace(':','-')}.jpg`; photoPriseUrl = uploadPhoto(data.photo, data.photoMime, filename); } catch(err) { Logger.log("Erreur upload photo prise : " + err.message); photoPriseUrl = "Erreur upload"; } } const convoyeur = (data.convoyeur || "").toUpperCase(); sheet.appendRow([ id, dateStr, heureStr, plaque, data.format || "SIV", data.client || "", // col 6 : Client statutPrise, // col 7 : Statut prise photoPriseUrl, // col 8 : Photo prise (lien Drive) data.note || "", // col 9 : Note prise "EN COURS", // col 10 : Statut clôture (provisoire) "", // col 11 : Note clôture "", // col 12 : Photo clôture "", // col 13 : Heure clôture "", // col 14 : Coût convoyeur // col 15 : Convoyeur ]); // Coloriser la ligne selon statut prise const lastRow = sheet.getLastRow(); const couleur = statutPrise === "PROBLÈME" ? "#FFF3E0" : "#FFFFFF"; sheet.getRange(lastRow, 1, 1, 14).setBackground(couleur); return { ok: true, id: id, date: dateStr, heure: heureStr, statutPrise: statutPrise }; } // ============================================================ // CLÔTURER UN CONVOYAGE // ============================================================ function closeConvoyage(data) { const sheet = initSheet(); const now = new Date(); const tz = Session.getScriptTimeZone(); const heureStr = Utilities.formatDate(now, tz, "HH:mm"); const dateStr = Utilities.formatDate(now, tz, "dd/MM/yyyy"); const values = sheet.getDataRange().getValues(); for (let i = 1; i < values.length; i++) { if (values[i][0] === data.id) { const statutCloture = data.ok ? "OK" : "PROBLÈME"; const plaque = values[i][3] || "photo"; let photoCloUrl = ""; // Upload photo clôture si fournie if (data.photo && data.photoMime) { try { const filename = `${plaque}_cloture_${dateStr.replace(/\//g,'-')}_${heureStr.replace(':','-')}.jpg`; photoCloUrl = uploadPhoto(data.photo, data.photoMime, filename); } catch(err) { Logger.log("Erreur upload photo clôture : " + err.message); photoCloUrl = "Erreur upload"; } } // Mise à jour colonnes 10 à 14 // Coût : 3€ uniquement si le convoyeur est Paul (col 15 = index 14) const convoyeurClo = (values[i][14] || "").toUpperCase(); const cout = (convoyeurClo === "PAUL") ? TARIF : 0; sheet.getRange(i + 1, 10).setValue(statutCloture); // Statut clôture sheet.getRange(i + 1, 11).setValue(data.note || ""); // Note clôture sheet.getRange(i + 1, 12).setValue(photoCloUrl); // Photo clôture sheet.getRange(i + 1, 13).setValue(heureStr); // Heure clôture sheet.getRange(i + 1, 14).setValue(cout); // Coût // Coloriser selon le statut le plus sévère (prise OU clôture) const statutPrise = values[i][6]; // col 7 = index 6 let couleur = "#E8F5E9"; // vert par défaut (tout OK) if (statutPrise === "PROBLÈME" && !data.ok) couleur = "#FFEBEE"; // rouge : 2 problèmes else if (statutPrise === "PROBLÈME" || !data.ok) couleur = "#FFF3E0"; // orange : 1 problème sheet.getRange(i + 1, 1, 1, 14).setBackground(couleur); return { ok: true, heure: heureStr, tarif: cout, photoUrl: photoCloUrl }; } } return { error: "Convoyage introuvable (ID: " + data.id + ")" }; } // ============================================================ // RAPPORT DU JOUR // ============================================================ function formatCellDate(val, tz) { if (!val) return ''; if (val instanceof Date) return Utilities.formatDate(val, tz, "dd/MM/yyyy"); return String(val); } function formatCellTime(val, tz) { if (!val) return ''; if (val instanceof Date) return Utilities.formatDate(val, tz, "HH:mm"); return String(val); } function getRapportJour(dateParam) { const sheet = initSheet(); const values = sheet.getDataRange().getValues(); const tz = Session.getScriptTimeZone(); const today = dateParam || Utilities.formatDate(new Date(), tz, "dd/MM/yyyy"); const convoyages = []; let total = 0; for (let i = 1; i < values.length; i++) { const dateCell = formatCellDate(values[i][1], tz); if (dateCell === today) { const cout = parseFloat(values[i][13]) || 0; total += cout; convoyages.push({ id: values[i][0], date: dateCell, heurePrise: formatCellTime(values[i][2], tz), plaque: values[i][3], format: values[i][4], client: values[i][5], statutPrise: values[i][6], photoPrise: values[i][7], notePrise: values[i][8], statutCloture: values[i][9], noteCloture: values[i][10], photoCloture: values[i][11], heureCloture: formatCellTime(values[i][12], tz), cout: cout, convoyeur: values[i][14] || "" }); } } return { date: today, convoyages, total, count: convoyages.length, tarif: TARIF }; } // ============================================================ // RAPPORT MENSUEL // ============================================================ function getRapportMois(mois, annee) { const sheet = initSheet(); const values = sheet.getDataRange().getValues(); const tz = Session.getScriptTimeZone(); const prefix = `${String(mois).padStart(2,'0')}/${annee}`; const convoyages = []; let total = 0; for (let i = 1; i < values.length; i++) { const date = String(values[i][1]); const dateCell = formatCellDate(values[i][1], tz); if (dateCell.endsWith(prefix)) { const cout = parseFloat(values[i][13]) || 0; total += cout; convoyages.push({ id: values[i][0], date: dateCell, heurePrise: formatCellTime(values[i][2], tz), plaque: values[i][3], format: values[i][4], client: values[i][5], statutPrise: values[i][6], photoPrise: values[i][7], notePrise: values[i][8], statutCloture: values[i][9], noteCloture: values[i][10], photoCloture: values[i][11], heureCloture: formatCellTime(values[i][12], tz), cout: cout, convoyeur: values[i][14] || "" }); } } return { mois, annee, convoyages, total, count: convoyages.length, tarif: TARIF }; } // ============================================================ // HELPER // ============================================================ function jsonResponse(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
